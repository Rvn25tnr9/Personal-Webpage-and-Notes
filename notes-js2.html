function reverseString(reverseMe){
  var reversed ="";
  for (var i = reverseMe.length - 1; i >= 0; i--){
    reversed += reverseMe[i];
  }
  return reversed;
}

Building Triangle

function makeLine(length) {
    var line = "";
    for (var j = 1; j <= length; j++) {
        line += "* ";
    }
    return line + "\n";
    console.log(line);
}

function buildTriangle(row) {
  var triangle = '';
  for (var n = 1; n <= row; n++) {
    triangle += makeLine(n);
  }
  return triangle;
}
console.log(buildTriangle(10));
