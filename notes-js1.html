  function randomNumber() {
      var n = Math.random();
      var randomizer = Math.floor((100 * n) + 1);
      return randomizer;
  }

  function loveScore() {
      if (randomizer > 70) {
          console.log("Your love Score is " + randomizer + " %");

      }
  }
